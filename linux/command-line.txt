# Change directory permissions (755) recursively
find <target> -type d -exec chmod 755 {} \;

---

# Change file permissions (644) recursively
find <target> -type f -exec chmod 644 {} \;

---

# Copy files recursively without preserving attributes
cp --recursive --no-preserve=all --verbose <source> <target>

---

# Replace lowercase space and underscore with dashes
rename 'y/A-Z/a-z/; s/[ _]+/-/g' <target>

---

# Monitor memory usage for dirty and writeback pages
watch -d -n1 grep -e Dirty -e Writeback /proc/meminfo

---

# Change ownership recursively
chown --recursive <user>:<group> <target>

---

# Set scrolling speed for touchpad
xinput --set-prop <id> "libinput Scrolling Pixel Distance" <value>

---

# Sync file update if the size is different and preserve all
sudo rsync -avhi --progress --size-only /source/ /destination/

---
